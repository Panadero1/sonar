use serde::{Deserialize, Serialize};

use crate::{
    utility::animation::Animation,
    world::{
        space::GamePos,
        tile::{get_default_anim, match_directions, Tile, TileVariant},
    },
};

#[derive(Debug, Serialize, Deserialize)]
pub struct !TileName {
    pos: GamePos,
    anim: Animation,
    !anything else?
}

#[typetag::serde]
impl Tile for !TileName {
    fn get_pos(&self) -> GamePos {
        self.pos
    }

    fn get_anim_mut(&mut self) -> &mut Animation {
        &mut self.anim
    }

    fn next(&self) -> Option<Box<dyn Tile>> {
        Some(Box::new(!NextTile::new(!params)))
    }

    fn create(&self, pos: GamePos, variant: TileVariant) -> Box<dyn Tile> {
        Box::new(!TileName::new(pos, variant))
    }

    fn pick_tile(&self) -> Box<dyn Tile> {
        Box::new(Self {
            pos: (0, 0).into(),
            anim: get_default_anim((0, 0)),
        })
    }
}

impl !TileName {
    pub fn new(pos: GamePos, direction: TileVariant) -> !TileName {
        !TileName {
            pos,
            anim: get_default_anim(match_directions(direction, !spritesheet_coordinates)),
        }
    }
}
